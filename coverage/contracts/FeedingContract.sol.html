pragma solidity >=0.6.6 <0.9.0;
 
import '@uniswap/v2-periphery/contracts/interfaces/IUniswapV2Router02.sol';
import '@openzeppelin/contracts/token/ERC20/IERC20.sol';
 
contract Feeding {
    address private constant UNISWAP_V2_ROUTER = 0x7a250d5630B4cF539739dF2C5dAcb4c659F2488D; // Uniswap V2 Router address
 
    struct Vesting {
        uint256 amount;
        uint256 vestingTime;
    }
 
    mapping(address => Vesting) public vestingBalances;
 
    constructor() public {
    }
 
    function swapTokens(address _tokenIn, address _tokenOut, uint256 _amountIn, uint256 _amountOutMin, uint256 _vestingTime) external {
        IERC20(_tokenIn).transferFrom(msg.sender, address(this), _amountIn);
        IERC20(_tokenIn).approve(UNISWAP_V2_ROUTER, _amountIn);
 
        address[] memory path;
        if (_tokenIn != address(0xC02aaA39b223FE8D0A0e5C4F27eAD9083C756Cc2) && _tokenOut != address(0xC02aaA39b223FE8D0A0e5C4F27eAD9083C756Cc2)) {
            path = new address[](3);
            path[0] = _tokenIn;
            path[1] = 0xC02aaA39b223FE8D0A0e5C4F27eAD9083C756Cc2; // WETH Address
            path[2] = _tokenOut;
        } else {
            path = new address[](2);
            path[0] = _tokenIn;
            path[1] = _tokenOut;
        }
 
        // Make the swap
        uint[] memory amounts = IUniswapV2Router02(UNISWAP_V2_ROUTER).swapExactTokensForTokens(
            _amountIn,
            _amountOutMin,
            path,
            address(this), // Tokens are stored in this contract
            block.timestamp
        );
 
        // Store vesting balance
        vestingBalances[msg.sender] = Vesting({
            amount: amounts[amounts.length - 1],
            vestingTime: block.timestamp + _vestingTime
        });
    }
 
    function claim(address _tokenOut) external {
        require(block.timestamp >= vestingBalances[msg.sender].vestingTime, "Vesting period not reached");
        uint256 amount = vestingBalances[msg.sender].amount;
        require(amount > 0, "Nothing to claim");
 
        // Reset vesting balance
        delete vestingBalances[msg.sender];
 
        // Transfer tokens
        IERC20(_tokenOut).transfer(msg.sender, amount);
    }
 
    function claimable(address account) external view returns (uint256 amount) {
        if(block.timestamp < vestingBalances[account].vestingTime) {
            amount = vestingBalances[account].amount;
        }
        else {
            amount = 0;
        }
    }
 
    function timeLeft(address account) external view returns (uint256 timeLeft) {
        if(block.timestamp < vestingBalances[account].vestingTime) {
            timeLeft = vestingBalances[account].vestingTime - block.timestamp;
        }
        else {
            timeLeft = 0;
        }
    }
}
